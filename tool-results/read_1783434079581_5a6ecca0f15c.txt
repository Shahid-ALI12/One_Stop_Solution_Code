   312→  // Helper to calculate side-by-side PKT translation
   313→  const getPKTEquivalent = () => {
   314→    const hh = parseInt(simSelectedHour) || 0;
   315→    const mm = parseInt(simSelectedMinute) || 0;
   316→    
   317→    // We simulate offsets relative to PKT (UTC+5)
   318→    // EST is UTC-5 (offset = -10 hours compared to PKT)
   319→    // GMT/UTC is UTC+0 (offset = -5 hours compared to PKT)
   320→    // CET is UTC+1 (offset = -4 hours compared to PKT)
   321→    // PST is UTC-8 (offset = -13 hours compared to PKT)
   322→    let offsetFromPKT = -10; // default EST
   323→    if (simClientTz === 'GMT') offsetFromPKT = -5;
   324→    if (simClientTz === 'CET') offsetFromPKT = -4;
   325→    if (simClientTz === 'PST') offsetFromPKT = -13;
   326→
   327→    // Convert client time back to PKT: pkt_time = client_time - offsetFromPKT
   328→    let pktHour = (hh - offsetFromPKT) % 24;
   329→    if (pktHour < 0) pktHour += 24;
   330→
   331→    const ampm = pktHour >= 12 ? 'PM' : 'AM';
   332→    const displayHour = pktHour % 12 === 0 ? 12 : pktHour % 12;
   333→    const padMin = mm.toString().padStart(2, '0');
   334→
   335→    return `${displayHour}:${padMin} ${ampm} PKT (UTC+5)`;
   336→  };
